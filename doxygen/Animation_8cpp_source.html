00032 void Animation::Update(SDL_Rect& clip, const double delta_time) {
00033     // Compare the position on the sprite with the number of positions to know if is the
00034     // end of the animation.
00035     const bool k_end_of_animation = (CurrentFrame() >= m_number_of_images);
00036     const double k_time_for_each_frame = (m_total_time / m_number_of_images);
00037 
00038     m_total_elapsed_time += delta_time;
00039 
00040     // Check if the frame has changed.
00041     if(m_total_elapsed_time >= k_time_for_each_frame) {
00042         m_total_elapsed_time = 0;
00043         m_animation_count += 1;
00044 
00045         // Apparently defines the max horizontal sprites in the spritesheet.
00047         const int k_animation_limit = 10;
00048         assert(k_animation_limit >= 0 && "Must be >= 0");
00049 
00050         if(m_animation_count <= m_number_of_images) {
00051             if(m_x < k_animation_limit) {
00052                 m_x += 1;
00053             }
00054             else {
00055                 m_y += 1;
00056                 m_x = 0;
00057             }
00058         }
00059 
00060         if(k_end_of_animation) {
00061             m_x = m_initial_x;
00062             m_y = m_initial_y;
00063             m_animation_count = 0;
00064         }
00065     }
00066 
00067     const int k_position_x = m_x * m_sprite_width;
00068     const int k_position_y = m_y * m_sprite_height;
00069     UpdateClip(clip, k_position_x, k_position_y);
