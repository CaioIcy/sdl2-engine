66  this->currentState->load();
67 
68  // Get the first game time.
69  double totalGameTime = 0.0;
70  const double deltaTime = 1.0 / 60.0;
71  double accumulatedTime = 0.0;
72 
73  // This is the main game loop.
74  while(this->isRunning) {
75 
76  const double frameTime = FPSWrapper::delay(this->fpsManager);
77  accumulatedTime += frameTime;
78 
79  // Update.
80  while(accumulatedTime >= deltaTime) {
81  this->inputHandler->handleInput();
82 
83  // Check for an exit signal from input.
84  if(this->inputHandler->isQuitFlag() == true) {
85  stop();
86  return;
87  }
88 
89  this->currentState->update(deltaTime);
90 
91  accumulatedTime -= deltaTime;
92  totalGameTime += deltaTime;
93  }
94 
95  // Render.

97 
98  this->currentState->render();
99 
100  window->render();
101 
102  }
